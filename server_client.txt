server:
signal:1_to_2
slots:

client1:
GobangBoard instance1(gobangboard1):
signal:requestPlay(int x,int y)
slot:updateInformation(int stateCopy,int playCopy,int boardCopy[SIZE][SIZE])
BoardUi instance1(boardui1):
signal:boardChange(int state,int player,int board[SIZE][SIZE])
slot:play(int x,int y)

client2:
GobangBoard instance2(gobangboard2):
signal:requestPlay(int x,int y)
slot:updateInformation(int stateCopy,int playCopy,int boardCopy[SIZE][SIZE])
BoardUi instance2(boardui2):
signal:boardChange(int state,int player,int board[SIZE][SIZE])
slot:play(int x,int y)

以client1为sender来说要做以下连接(这里是把server当透明来看,实际上连接不是这样的,也就是说其中receiver是2的情况下,要receiver实际要改为server,server端至少要设置两种信号和两个槽)
QObject::connect(&gobangboard1,SIGNAL(requestPlay(int,int)),&boardui1,SLOT(play(int,int)))
QObject::connect(&gobangboard1,SIGNAL(requestPlay(int,int)),&boardui2,SLOT(play(int,int)))
QObject::connect(&boardui1,SIGNAL(boardChange(int,int,int [SIZE][SIZE])),&gobangboard1,SLOT(updateInformation(int,int,int [SIZE][SIZE])))
QObject::connect(&boardui1,SIGNAL(boardChange(int,int,int [SIZE][SIZE])),&gobangboard2,SLOT(updateInformation(int,int,int [SIZE][SIZE])))
